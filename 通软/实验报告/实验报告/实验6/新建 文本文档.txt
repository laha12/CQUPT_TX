#include <Windows.h>
#include <stdio.h>
#include <stdlib.h>

// 定义一个函数，用于打印图像的信息
void putpicture(BITMAPFILEHEADER fileHead, BITMAPINFOHEADER infoHead);

int main() {
    // 打开 BMP 文件
    FILE* fp = fopen("view_256.bmp", "rb");
    if (fp == NULL) {
        printf("文件打开失败\n");
        return 0;
    }

    // 读取 BMP 文件头和信息头
    BITMAPFILEHEADER fileHead;
    BITMAPINFOHEADER infoHead;
    fread(&fileHead, sizeof(BITMAPFILEHEADER), 1, fp);
    fread(&infoHead, sizeof(BITMAPINFOHEADER), 1, fp);

    // 提取图像宽度、高度、位深等信息
    int width = infoHead.biWidth;
    int height = infoHead.biHeight;
    int biCount = infoHead.biBitCount;
    int lineByte = (width * biCount / 8 + 3) / 4 * 4;  // 计算每行的字节数

    // 申请颜色表内存并读取
    RGBQUAD* pColorTable;
    pColorTable = (RGBQUAD*)malloc(256 * sizeof(RGBQUAD));
    fread(pColorTable, sizeof(RGBQUAD), 256, fp);

    // 申请图像数据内存并读取
    unsigned char* pBmpBuf;
    pBmpBuf = (unsigned char*)malloc(lineByte * height);
    fread(pBmpBuf, lineByte * height, 1, fp);
    fclose(fp);

    printf("旋转前图片的参数为：\n");
    putpicture(fileHead, infoHead);
    printf("\n");

    // 左旋转90度
    unsigned char* pBmpBuf3;
    int lineByte3 = (height * biCount / 8 + 3) / 4 * 4;

    pBmpBuf3 = (unsigned char*)malloc(lineByte3 * width);
    for (int i = 0; i < width; ++i) {
        for (int j = 0; j < height; ++j) {
            unsigned char* p;
            p = (unsigned char*)(pBmpBuf3 + lineByte3 * i + j);
            (*p) = 255;  // 设置新图像的像素值为白色
        }
    }
    for (int i = 0; i < width; ++i) {
        for (int j = 0; j < height; ++j) {
            unsigned char* p, * p3;
            p = (unsigned char*)(pBmpBuf + lineByte * j + i);  // 原图像
            p3 = (unsigned char*)(pBmpBuf3 + lineByte3 * i + height - j - 1);  // 新图像
            (*p3) = (*p);  // 将原图像的像素复制到新图像中
        }
    }

    // 创建并写入旋转后的 BMP 文件
    FILE* fpo = fopen("view_256(2).bmp", "wb");
    if (fpo == NULL)
        return 0;

    fwrite(&fileHead, sizeof(BITMAPFILEHEADER), 1, fpo);

    BITMAPINFOHEADER infoHead3;
    infoHead3.biBitCount = biCount;
    infoHead3.biClrImportant = 0;
    infoHead3.biClrUsed = 0;
    infoHead3.biCompression = 0;
    infoHead3.biHeight = width;  // 交换宽度和高度
    infoHead3.biPlanes = 1;
    infoHead3.biSize = 40;
    infoHead3.biSizeImage = lineByte3 * width;
    infoHead3.biWidth = height;
    infoHead3.biXPelsPerMeter = 0;
    infoHead3.biYPelsPerMeter = 0;

    fwrite(&infoHead3, sizeof(BITMAPINFOHEADER), 1, fpo);
    fwrite(pColorTable, sizeof(RGBQUAD), 256, fpo);
    fwrite(pBmpBuf3, lineByte3 * width, 1, fpo);
    fclose(fpo);

    printf("旋转后图片的参数为：\n");
    putpicture(fileHead, infoHead3);

    // 释放内存
    free(pColorTable);
    free(pBmpBuf);
    free(pBmpBuf3);

    return 1;
}

// 打印图像信息的函数
void putpicture(BITMAPFILEHEADER fileHead, BITMAPINFOHEADER infoHead) {
    printf(" bfSize: %u\n bfoffBits: %u\n ", fileHead.bfSize, fileHead.bfOffBits);
    printf("biWidth: %ld\n biHeight: %ld\n ", infoHead.biWidth, infoHead.biHeight);
    printf("biBitCount: %u\n biPlanes: %u\n ", infoHead.biBitCount, infoHead.biPlanes);
}
