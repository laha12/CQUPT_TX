#include<winsock2.h>
#include<iostream>
#include<string>
#include<stdio.h>
using namespace std;
#pragma comment(lib,"ws2_32.lib")

int main() {

	//请求协议版本
	WORD sockVersion = MAKEWORD(2, 2);//显示版本，2.2版本
	WSADATA wsdata;//wWSADATA是一个结构体
	WSAStartup(sockVersion, &wsdata);
	if (LOBYTE(wsdata.wVersion) != 2 || HIBYTE(wsdata.wVersion) != 2)
	{
		printf("请求协议失败\n");
		return -1;
	}
	printf("请求协议成功\n");

	//创建套接字
	SOCKET serverSocket = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);//AF_INET通信协议类型, SOCK_STREAM通讯数据类型, IPPROTO_TCP保护方式
	if (serverSocket == INVALID_SOCKET)
	{
		printf("创建Socket失败\n");
		WSACleanup();
		return -2;
	}
	printf("创建Socket成功\n");

	//绑定套接字
	sockaddr_in sockAddr = { 0 };
	sockAddr.sin_family = AF_INET;
	sockAddr.sin_addr.S_un.S_addr = INADDR_ANY;//绑定地址
	sockAddr.sin_port = htons(9999);//端口号
	if (bind(serverSocket, (sockaddr*)&sockAddr, sizeof(sockAddr)) == SOCKET_ERROR) {
		printf("绑定失败\n");
		WSACleanup();
		return -1;
	}
	printf("绑定成功\n");


	//开始监听
	//可能有多个客户端连接，10表示队列的长度
	if (listen(serverSocket, 10) == SOCKET_ERROR) {
		printf("监听失败\n");
		WSACleanup();
		return -1;
	}
	printf("开启监听成功\n");


	//等待客户端连接     阻塞   尾声抱柱
	SOCKADDR_IN cAddr = { 0 };
	int len = sizeof(cAddr);
	SOCKET clientSocket;

	char msg[1024];//存储传送的消息
	int flag = 0;//是否已经连接上
	printf("等待连接...\n");
	clientSocket = accept(serverSocket, (sockaddr*)&cAddr, &len);
	if (SOCKET_ERROR == clientSocket)
	{
		printf("服务器宕机\n");
		WSACleanup();
		return -1;
	}
	printf("有客户连接到服务器：%s\n!", inet_ntoa(cAddr.sin_addr));

	//通信
	while (1)
	{
		int r = recv(clientSocket, msg, 1023, NULL);
		if (r > 0) {
			msg[r] = 0;//添加'\0'
			printf("%s\n", msg);
		}
	}
	closesocket(clientSocket);
	WSACleanup();






	return 0;
}
