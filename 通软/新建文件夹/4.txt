#include <Windows.h>
#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include<string.h>

#define pai 3.1415926

int main()
{
    //以二进制读取bmp文件
    FILE *fp;
    if((fp=fopen("E:\\通信软件基础\\4图片旋转.bmp", "rb"))==0)
    {
        printf("错误\n");
        exit(0);
        }

//用fileHeader,infoHeader来存储位图文件头和位图信息头
    BITMAPFILEHEADER fileHeader;
    BITMAPINFOHEADER infoHeader;

    fread(&fileHeader,sizeof(BITMAPFILEHEADER),1,fp);//读取位图文件头
    fread(&infoHeader,sizeof(BITMAPINFOHEADER),1,fp);//读取位图信息头

    int width=infoHeader.biWidth;
    int highth=infoHeader.biHeight;
    int bicount=infoHeader.biBitCount;
    int linebyte=(width*bicount/8+3)/4*4;//计算一行的字节数

    RGBQUAD *pcolortable;

    pcolortable=new RGBQUAD[256];//申请一个空间来存储调色板
    fread(pcolortable,sizeof(RGBQUAD),256,fp);

    unsigned char *pBmpBufsrc,*pBmpBufDst;
    pBmpBufsrc=new unsigned char[linebyte*highth];
    fread(pBmpBufsrc,linebyte*highth,1,fp);//存储源图像的位图数据

    fclose(fp);

    //输入想要的角度
    printf("请输入想要的角度： ");
    double angle;
    scanf("%lf",&angle);


    //换算要旋转的角度为弧度制
    double radian=angle*pai/180;

    //求四个角对应的像素数学坐标

	double x1 = -width/2;
	double y1 = highth/2;
	double x2 = width/2;
	double y2 = highth/2;
	double x3 = -width/2;
	double y3= -highth/2;
	double x4= width/2;
	double y4= -highth/2;


    //旋转之后四个顶点的坐标
	double sina = sin(radian);
	double cosa = cos(radian);
	double afterx1 = x1*cosa + y1*sina;
	double aftery1 = -x1*sina + y1*cosa;
	double afterx2 = x2*cosa + y2*sina;
	double aftery2 = -x2*sina + y2*cosa;
	double afterx3 = x3*cosa + y3*sina;
	double aftery3 = -x3*sina + y3*cosa;
	double afterx4 = x4*cosa + y4*sina;
	double aftery4 = -x4*sina + y4*cosa;
//旋转后图像宽和高
	int wide = fmax(abs(afterx2-afterx3),abs(afterx1 - afterx4));
	int high = fmax(abs(aftery2 - aftery3),abs(aftery1- aftery4));

    int lineByte1 = (wide*bicount / 8 + 3) / 4 * 4;
	pBmpBufDst=new unsigned char[lineByte1*high];

	//将pBmpBufDst里面的值初始为255，即为白色
	for (int i = 0; i < high; ++i){
		for (int j = 0; j < wide; ++j){
			unsigned char *p;
			p = (unsigned char *)(pBmpBufDst + i*lineByte1 + j);
			(*p) = 255;
		}
	}

for (int i = 0; i < high; i++) {
		for (int j = 0; j < wide; j++) {
//转换到以图像为中心的坐标系，并进行逆旋转
			int x = (j - wide / 2)*cos((360 - angle)*pai/180) -(-i + high / 2)*sin((360 - angle)*pai/180);
			int y = (j - wide / 2)*sin((360 - angle)*pai/180) + (-i + high / 2)*cos((360 - angle)*pai/180);
//如果这个坐标不在原图像内，则不赋值
			if (x > width / 2 || x < -width / 2 || y > highth / 2 || y < -highth / 2) {
				continue;
			}
//再转换到原图像坐标系下
			int tXN = x+ width / 2;
			int tYN = abs(y - highth / 2);
//值拷贝，3是一个位图数据占的字节
			memcpy(&pBmpBufDst[i * lineByte1 + j * bicount / 8],&pBmpBufsrc[tYN * linebyte + tXN * bicount / 8],3);
		}
	}


	char *name="newphoto.bmp";//图片名
	FILE *fpc=fopen(name,"wb");
	if (fpc==0)return 0;
	int colortablesize=0;
	if(bicount==8)colortablesize=1024;//4*256,灰度图；
	BITMAPFILEHEADER dstFileheader;//定义目标文件头
	dstFileheader.bfOffBits=54+colortablesize;//bfOffBits是图像文件前三个部分所需空间之和
	dstFileheader.bfReserved1=0;
	dstFileheader.bfReserved2=0;
	dstFileheader.bfSize= sizeof(BITMAPFILEHEADER) + sizeof(BITMAPINFOHEADER)+ colortablesize + lineByte1*high;//bfSize是图像文件4个组成部分之和
	dstFileheader.bfType=0x4D42	;//bmp
	fwrite(&dstFileheader,sizeof(dstFileheader),1,fpc);
//申请位图信息头结构变量，填写信息头信息
	BITMAPINFOHEADER dstInfoheader;
	dstInfoheader.biBitCount=bicount;//同上
	dstInfoheader.biClrImportant=0;
	dstInfoheader.biClrUsed=0;
	dstInfoheader.biCompression=0;
	dstInfoheader.biHeight=high;
	dstInfoheader.biPlanes=1;
	dstInfoheader.biSize=40;
	dstInfoheader.biSizeImage=lineByte1*high;//新的位图数据需要的字节
	dstInfoheader.biWidth=wide;
	dstInfoheader.biXPelsPerMeter=0;
	dstInfoheader.biYPelsPerMeter=0;
	//写位图信息头进内存
	fwrite(&dstInfoheader,sizeof(BITMAPINFOHEADER),1,fpc);
//如果灰度图像,有颜色表,写入文件
if(bicount==8)
    fwrite(pcolortable, sizeof(RGBQUAD),256, fpc);
//写位图数据进文件
    fwrite(pBmpBufDst,high*lineByte1,1,fpc);

    fclose(fpc);
    system("pause");
    return 1;


}
