void Distant(float A[3],float B[3],float C[3])
{
float dis[15][3]={0};
    for(lines=0;lines<15;lines++)
    {

        for(j=0;j<3;j++)
        {
            dis[lines][0]+=fabs(origin[lines][j]-A[j]);
            dis[lines][1]+=fabs(origin[lines][j]-B[j]);
            dis[lines][2]+=fabs(origin[lines][j]-C[j]);
        }
        if(dis[lines][0]<dis[lines][1]&&dis[lines][0]<dis[lines][2])level[lines]=0;
        if(dis[lines][1]<dis[lines][2]&&dis[lines][1]<dis[lines][0])level[lines]=1;
        if(dis[lines][2]<dis[lines][1]&&dis[lines][2]<dis[lines][0])level[lines]=2;

    }

 }



void new(int level[15])

//求相同等级的数据和
    for(lines=0;lines<15;lines++)
    {

    if(level[lines]==1){M[y]=lines;  y++;}
    if(level[lines]==2){L[z]=lines;  z++;}
        for(i=0;i<3;i++)
        {
            New[level[lines]][i]+=origin[lines][i] ;

    }


}
for(i=0;i<3;i++)

        New[1][i]=New[1][i]/y;

for(j=0;j<3;j++)
{
       New0[j]=New[0][j];
       New1[j]=New[1][j];
       New2[j]=New[2][j];
}

}
