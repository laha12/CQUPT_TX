#### 静态路由

R0

```cisco
enable
configure terminal
! 配置接口Fa0/0的IP地址（连接192.168.1.0/24网段）
interface FastEthernet0/0
 ip address 192.168.1.254 255.255.255.0
 no shutdown
! 配置接口Fa0/1的IP地址（连接192.168.3.0/24网段）
interface FastEthernet0/1
 ip address 192.168.3.1 255.255.255.0
 no shutdown
! 配置到192.168.2.0/24网段的静态路由，下一跳指向路由器1的Fa0/1接口（192.168.3.2）
ip route 192.168.2.0 255.255.255.0 192.168.3.2
end
write memory
```

左侧二层交换机 接入PC0&1,流通数据给路由器1

```
enable
configure terminal
#创建vlan
vlan 190
#命名vlan
name VLAN190
vlan 290
name VLAN290

interface FastEthernet0/1
#将模式改为access
 switchport mode access
#接入vlan190
 switchport access vlan 190
 #开启端口
 no shutdown

interface FastEthernet0/2
 switchport mode access
 switchport access vlan 290
 no shutdown
 
 #接入路由器 trunk类型
interface FastEthernet0/3
 switchport mode trunk
 #运行vlan通过
 switchport trunk allowed vlan 190,290
 no shutdown
```



单臂路由——路由器 实现PC0&1通信

```
enable
configure terminal
interface FastEthernet0/0
 no shutdown
#配置VLAN190的子接口
interface FastEthernet0/0.190
 encapsulation dot1Q 190
 ip address 10.90.10.1 255.255.255.0  #（网关IP+子网掩码）
#配置VLAN290的子接口
interface FastEthernet0/0.290
 encapsulation dot1Q 290
 ip address 20.90.20.1 255.255.255.0
#配置互联接口Fa0/1
interface FastEthernet0/1
 ip address 3.90.3.1 255.255.255.0
 no shutdown
```

右侧二层交换机 接入PC2&3,流通数据给三层交换机

```
enable
configure terminal
#创建vlan
vlan 190
#命名vlan
name VLAN190
vlan 290
name VLAN290

 #开启端口

```



三层交换机 实现PC2&3之间通信 

```
enable
configure terminal

ip routing#使三层交换机具有路由器的功能

#创建vlan
vlan 190
#命名vlan
name VLAN190
vlan 290
name VLAN290



#配置SVI接口（VLAN的三层网关） 设置vlan190网关
interface Vlan190
 ip address 100.90.100.1 255.255.255.0
 no shutdown

interface Vlan290
 ip address 200.90.200.1 255.255.255.0
 no shutdown

#互联接口Fa0/2
interface FastEthernet0/2
#开启配置接口IP功能
 no switchport
 ip address 3.90.3.2 255.255.255.0
 no shutdown
```
